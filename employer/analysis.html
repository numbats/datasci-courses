out <- df %>% 
  mutate(salary = map_dbl(salary_string, ~{
    out <- str_extract_all(.x, "[0-9]+,?[0-9]+[kK.%]?") %>% 
      str_replace("[,+]", "") %>% 
      str_replace("[kK]", "000")
    out <- out[!out %in% "00"]
    out <- out[!str_detect(out, "[.%]")]
    if(length(out)==0) return(NA_real_)
    x <- mean(as.numeric(out))
    if(!is.na(.x) && any(str_detect(.x, "(day|pd)"))) return(x * 48 * 5)
    return(x)
  })) 
money <- out %>%
